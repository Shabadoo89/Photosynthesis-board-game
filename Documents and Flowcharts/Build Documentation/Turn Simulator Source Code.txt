#include <iostream>
#include <string>

using namespace std;

int main()
{
	bool troof = 1;
	int turn_counter = 0;
	int day_counter = turn_counter;

	do
	{
		cout << "This starts the turn simulator" << endl;

		bool stuff = 1;

		cout << "Player 1, take your turn: " << endl;
		do
		{
			char shit;
			cout << "Blah blah blah\n";
			cout << "Doing some stuff\n";
			cout << "Would you like to stop doing some stuff?\n";
			cin >> shit;

			if (shit == 'y' || shit == 'Y')
				stuff = 0;
			else
				cout << "Okay, let's keep doing stuff.\n";
		} while (stuff);


		stuff = 1;
		cout << "Player 2, take your turn: \n";
		do
		{
			char shit;
			cout << "Blah blah blah\n";
			cout << "Doing some stuff\n";
			cout << "Would you like to stop doing some stuff?\n";
			cin >> shit;

			if (shit == 'y' || shit == 'Y')
				stuff = 0;
			else
				cout << "Okay, let's keep doing stuff.\n";
		} while (stuff);

		if (turn_counter == 6)
		{
			day_counter++;
			turn_counter = 0;
		}
		else
			turn_counter++;

		if (day_counter == 3)
			troof = 0;

	} while (troof);

	cout << "This signifies the end of the game" << endl;
	cout << "Thank you for playing...Goodbye";

	system("pause");
	return 0;
}
